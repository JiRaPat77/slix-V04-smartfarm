#!/bin/sh

### BEGIN INIT INFO
# Provides:          sshtunnel1880
# Required-Start:    $network
# Required-Stop:
# Default-Start:     S
# Default-Stop:
# Short-Description: SSH Tunnel to weaverbase.io
### END INIT INFO

DAEMON="/usr/bin/ssh"
DAEMON_ARGS="-i /root/.ssh/id_ed25519 -NT -o StrictHostKeyChecking=no -o ServerAliveInterval=60 -o ServerAliveCountMax=3 -R slix005:1880:localhost:1880 -R slix005:22:localhost:22 slix005@weaverbase.io"
USER="root"
PIDFILE="/var/run/sshtunnel.pid"

start() {
    echo "Starting SSH tunnel service in background..."
   
    (
        echo "Enforcing Network Settings..."
        ifconfig eth0 192.168.1.100 netmask 255.255.255.0 up
        route add default gw 192.168.1.1 2>/dev/null
        echo "nameserver 8.8.8.8" > /etc/resolv.conf
        
        sleep 5
        
        while true; do
            echo "Connecting to Weaverbase..."
            /usr/bin/ssh -i /root/.ssh/id_ed25519 -NT -o StrictHostKeyChecking=no -o ServerAliveInterval=60 -o ServerAliveCountMax=3 -R slix005:1880:localhost:1880 -R slix005:22:localhost:22 slix005@weaverbase.io
            
            echo "SSH Tunnel disconnected or failed! Retrying in 10 seconds..."
            sleep 10
        done
    ) >> /tmp/ssh_tunnel.log 2>&1 &
    
 
    echo $! > /var/run/sshtunnel.pid
}

stop() {
    echo "Stopping SSH tunnel..."
    start-stop-daemon --stop --pidfile "$PIDFILE" --retry 5
    rm -f "$PIDFILE"

}

restart() {
    stop
    sleep 2
    start
}

case "$1" in
    start) start ;;
    stop) stop ;;
    restart) restart ;;
    *) echo "Usage: $0 {start|stop|restart}"; exit 1 ;;
esac

exit 0
